CREATE TABLE Blacklisted_Domains (
Domain STRING, 
LIST_NAME STRING,
CATEGORY        STRING,
COMMENTS        STRING,
SOURCE          STRING,
URL             STRING,
DESCRIPTION     STRING,
DATEADDED       TIMESTAMP,
DATEOFINFO      TIMESTAMP,
DATEEXPIRES     TIMESTAMP
) ROW FORMAT
DELIMITED FIELDS TERMINATED BY ‘|’
STORED AS TEXTFILE;
